function checkID(req, res, next) {
  var pupID = req.params.id;
  db.any('select id from pups where id = $1', pupID)
    .then(function(data) {
      if(!data.length) {
        /* jshint ignore:start */
        res.status(400)
          .json({
            status: 'error',
            message: `ID '${pupID}' does not exist.`
          });
        /* jshint ignore:end */
      } else {
        return next();
      }
    })
    .catch(function(err) {
      return next(err);
    });
}
