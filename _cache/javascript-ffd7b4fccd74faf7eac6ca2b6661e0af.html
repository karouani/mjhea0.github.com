function validPostObject(req, res, next) {
  var payload = req.body;
  var keys = ['name', 'breed', 'age', 'sex'];
  keys.forEach(function(key){
    if(!(key in payload)) {
      /* jshint ignore:start */
      return res.status(400)
        .json({
          status: 'error',
          message: `Invalid JSON. '${key}' does not exist.`
        });
      /* jshint ignore:end */
    }
  });
  return next();
}
