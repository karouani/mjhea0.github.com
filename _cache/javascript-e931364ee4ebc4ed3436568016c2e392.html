BookList.prototype.add = function(book) {
  this.bookArr.push(book);
  this.numNotRead += 1;

  if (this.curBook === "") {
    this.curBook = book;
  } else if (this.nextBook === "") {
    this.nextBook = book;
  }
}

BookList.prototype.finishCurrentBook = function() {
  this.numRead += 1;
  this.numNotRead -= 1;
  this.curBook.read = true;
  this.curBook.readDate = new Date(Date.now());
  this.lastBook = this.curBook;
  this.curBook = this.nextBook;
  this.nextBook = "";

  //Find next book to read, if any
  var idx = 0,
      foundBook = false;

  while (idx < this.bookArr.length && !foundBook) {
    var book = this.bookArr[idx];

    if (!book.read && book !== this.curBook) {
      this.nextBook = book;
      foundBook = true;
    }

    idx += 1;
  }
}
