// Constructors

var Book = function(attributes) {
  this.title = attributes.title;
  this.genre = attributes.genre;
  this.author = attributes.author;
  this.read = false;
  this.readDate = "";
}

var BookList = function() {

  this.numNotRead = 0;
  this.nextBook = "";
  this.curBook = "";
  this.lastBook = "";
  this.bookArr = [];
}
